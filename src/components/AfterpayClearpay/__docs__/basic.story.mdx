    /* collected from the buyer */
    billingContact: {
      addressLines: ['123 Main Street', 'Apartment 1'],
      familyName: 'Doe',
      givenName: 'John',
      countryCode: 'GB',
      city: 'London',
    },
    currencyCode: 'GBP',
    intent: 'CHARGE',
  })}
  locationId="<YOUT_LOCATION_ID>"
